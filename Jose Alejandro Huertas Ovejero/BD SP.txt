DELIMITER $$

USE `prueba`$$

CREATE FUNCTION `f_resta` (numero1 int,numero2 int)

RETURNS int

BEGIN
set @resul_resta=0;

select numero1-numero2 into @resul_resta;

RETURN @resul_resta;

END$$


DELIMITER ;

USE `prueba`;

DROP function IF EXISTS `f_multiplicar`;


DELIMITER $$

USE `prueba`$$

CREATE FUNCTION `f_multiplicar` (numero1 int,numero2 int)

RETURNS int

BEGIN

set @resul_mult=0;

select numero1*numero2  into @resul_mult;
RETURN @resul_mult;

END$$


DELIMITER ;

DELIMITER $$
USE `prueba`
$$
CREATE FUNCTION `f_dividir` (numero1 int,numero2 int)

RETURNS INT

BEGIN

set @resul_divid=0;

select numero1/numero2 into @resul_divid;

RETURN @resul_divid;

END
$$

DELIMITER ;
