	public class Operaciones{

		
		int a = 3, b = 6, res = 0;
	
		public void suma(){
			res = a + b;
			System.out.println("La suma es : " + res);	
		}


		public void resta(){
			res = a - b;
			System.out.println("La resta es : " + res);	
		}


		public void multiplicacion(){
			res = a * b;
			System.out.println("La multiplicación es : " + res);	
		}


		public void division(){
			if (b == 0){
				System.out.println("No se puede hacer la división");
			}
			else{	
				res = a / b;
				System.out.println("La división es : " + res);	
			}
		}

		
	}


	main(){
		Operaciones abrir = new Operaciones();
		abrir.suma();
		abrir.resta();
		abrir.multiplicacion();
		abrir.division();
	}

	public class Implementos{

		// i=minutos, j=segundos, k=milisegundos 
		
		public void cronometro()
		{
			for (int i = 0; i < 60; i++ )
			{
				for (int j = 0; j < 60; j++)
				{
					for (int k = 0; k < 100; k++)
					{
						System.out.println(i+":"+j+":"+k);
					}
				}
			}
		}


		public void temporizador()
		{

       			System.out.println ("Digite la hora:");

        		m=teclado.nextInt();
        		System.out.println ("Digite los segundos:");

        		for (int i = h; i >= 0; i--)
