CREATE DATABASE funciones

USE funciones

CREATE FUNCTION f_suma(@nota1 float, @nota2 float)
returns float
as
begin
declare @resultadoSum float;
set @resultadoSum = @nota1 + @nota2;
return @resultadoSum;
end

select dbo.f_suma(2, 3)


CREATE FUNCTION f_resta(@nota1 float, @nota2 float)
returns float
as
begin
declare @resultadoRest float
set @resultadoRest = @nota1 - @nota2
return @resultadoRest
end

select dbo.f_resta(2, 3)


CREATE FUNCTION f_produc(@nota1 float, @nota2 float)
returns float
as
begin
declare @resultadoProd float
set @resultadoProd = @nota1 * @nota2
return @resultadoProd
end

select dbo.f_produc(2, 3)


CREATE FUNCTION f_cociente(@nota1 float, @nota2 float)
returns float
as
begin
declare @resultadoCoc float		
set @resultadoCoc = @nota1 / @nota2
return ISNULL(@nota1 / NULLIF(@nota2, 0),0) 
	
end

select dbo.f_cociente(2, 0)


CREATE PROC spCalculadora

BEGIN

	DECLARE @resultadoSum float;
	DECLARE @resultadoRest float;
	DECLARE @resultadoProd float;
	DECLARE @resultadoCoc float;

	SET @resultadoSum = f_suma(3, 2);
	SET @resultadoRest = f_resta(3, 2);
	SET @resultadoProd = f_produc(3, 2);
	SET @resultadoCoc = f_cociente(3, 0);

END

EXEC spCalculadora
