import java.util.Scanner;

public class MedicionDeTiempo {
    public void cronometro( ){
      
        for( int min=0; min<30; min++){
            for(int seg=0; seg<30; seg++){
            System.out.println(min+":"+seg);
            }
        }
    }
    
    public void temporizador(int tiempo){

        int min=0;
        int seg=0;

        min=tiempo*60; 
        seg=min*60; 
    
        for( int i=0; i<tiempo; i++){
            for( int j=0; j<=min; j++){
                for(int k=0; k<=seg; k++){
               System.out.println(j+":"+k);
                }
            
            }
          
           System.out.println("ha pasado "+ min+" minutos " +seg+ " segundos, que equivalen a "+ tiempo+ " hora");
            
        }
  
    }

    public static void main( String[] args){
        Scanner x= new Scanner(System.in);
        MedicionDeTiempo medi= new MedicionDeTiempo();
	System.out.println("Cronometro 30 minutos");
        medi.cronometro();
        System.out.println("Por cuanto tiempo quieres poner el temporizador");
        System.out.print("horas: ");
        int tiempo= x.nextInt();
        medi.temporizador(tiempo);;
        
     
    }
