-- se crea la tabla estudiantes:
CREATE TABLE Estudiantes(

Id_Estudiante int NOT NULL,
Nombre varchar(60),
Apellido varchar(60),
Edad int,
Correo varchar (60),
Direccion varchar(60),
Telefono varchar(60),
PRIMARY KEY(Id_Estudiante)
--foreign key(Id_Profesor) references Profesores(Id_Profesor),
--foreign key(Id_Asignatura) references Asignatura (Id_Asignatura)

);

-- se crea la tabla Profesores:
CREATE TABLE Profesores(

Id_Profesor int,
Nombre varchar(60),
Apellido varchar(60),
Correo varchar (60),
Direccion varchar(60),
Telefono varchar(60),
PRIMARY KEY (Id_Profesor)
);

-- Se crea la tabla Asignatura:
CREATE TABLE Asignatura(

Id_Asignatura int,
Nombre varchar(60),
Horario date, 
Salon varchar(20),
Id_Estudiante int,
Id_Nota int,
Id_Profesor int,
PRIMARY KEY (Id_Asignatura),
foreign key (Id_Nota) references Notas (Id_Nota),
foreign key(Id_Profesor) references Profesores(Id_Profesor),
foreign key(Id_Estudiante) references Estudiantes (Id_Estudiante)

);

-- se crea la tabla de notas:
CREATE TABLE Notas(

Id_Nota int,
Id_Profesor int,
Nota_Parcial float,
Nota_Taller float,
Nota_Quiz float,
Nota_Final float,
PRIMARY KEY (Id_Nota),
foreign key(Id_Profesor) references Profesores(Id_Profesor)
);
