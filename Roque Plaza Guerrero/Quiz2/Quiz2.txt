import java.util.ArrayList;
import java.util.Scanner;

public class Main {
    static Scanner scn = new Scanner(System.in);

    public static void main(String[] args) {
        ArrayList<compras> listacompras = new ArrayList<>();
        int opcion = 0;

        do {
           opcion = opciones();


            switch (opcion) {
                case 1: //Agregar
                    System.out.println("----Agrega  Productos a tu carrito :) ---");
                    listacompras.add(agregarcompras());
                    break;


                case 2://ver  precios
                    System.out.println("----- Aqui tienes los precios de los productos en el carrito: ");
                    for (compras g : listacompras) {
                        System.out.println(g.getPrecio());
                    }



                   break;
                 case 3:// ver por categoria
                    System.out.println("--- En tu carrito tienes productos de estas categorias: ");
                     for (compras k : listacompras) {
                         System.out.println(k.getCategoria());
                     }


                    break;
                case 4:// ver por nombre
                    System.out.println("----- Estos son los  nombres de tus productos: ");
                    for (compras h : listacompras) {
                        System.out.println(h.getNombre());
                    }
                    break;
                case 5:

                    System.out.println("---------------------");
                    System.out.println("Estos son todos tus productos Agregados en el carrito: ");
                    break;

            }
        }while (opcion!=5);

        //Mostrar todas las compras
        for (compras e : listacompras) {
            System.out.println(e.mostrarDatos());
        }
        System.out.println("Gracias por usar el programa :)");

    }
    public static compras agregarcompras(){
        compras nuevacompra = new compras(""," ", 0);
        String nombre; String categoria; int precio;

        System.out.print("\nEscribe el nombre de producto: ");
        nombre = scn.next();
        nuevacompra.setNombre(nombre);

        System.out.print("\nEscribe la categoria: ");
        categoria = scn.next();
        nuevacompra.setCategoria(categoria);

        System.out.print("\nEscribe el precio: ");
        precio= scn.nextInt();
        nuevacompra.setPrecio(precio);


        return nuevacompra;
    }
    public static int opciones () {
                int opcion;
                System.out.println("----- Opciones -----");
                System.out.println("1. Agregar productos");
                System.out.println("2. estos son los precios");
                System.out.println("3. Estas son las categorias");
                System.out.println("4. productos por nombres ");
                System.out.println("5. Salir");
                System.out.println("-----------------");
                System.out.print("Escribe la opcion:");
                opcion = scn.nextInt();

                return opcion;
            }

        }
