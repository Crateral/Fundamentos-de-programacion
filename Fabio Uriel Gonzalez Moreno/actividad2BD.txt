CREATE database Calculadora
use database Calculadora

CREATE  function fnc_suma(valor1,valor2 int)
return int

BEGIN
	
	declare resultado int;
	SET @resultado = @valor1 + @valor2;
	return @resultado;
END

CREATE  function fnc_resta(valor1,valor2 int)
return int

BEGIN
	
	declare resultado int;
	SET @resultado = @valor1 - @valor2;
	return @resultado;
END

CREATE  function fnc_multiplicacion(valor1,valor2 int)
return int

BEGIN
	
	declare resultado int;
	SET @resultado = @valor1 * @valor2;
	return @resultado;
END

CREATE  function fnc_division(valor1,valor2 int)
return float

BEGIN
	
	declare resultado float;
	SET @resultado = @valor1 / @valor2;
	return @resultado;
END

CREATE procedure sp_Resultados

BEGIN
	
	DECLARE @ResultadoSuma int;
	DECLARE @ResultadoResta int;
	DECLARE @ResultadoMultiplicacion int;
	DECLARE @ResultadoDivision float;

	set @ResultadoSuma = fnc_suma(5,1);
	set @ResultadoSuma = fnc_resta(5,1);
	set @ResultadoSuma = fnc_multiplicacion(5,1);
	set @ResultadoSuma = fnc_division(5,1);

	
END
