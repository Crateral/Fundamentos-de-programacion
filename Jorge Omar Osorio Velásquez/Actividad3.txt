//Clase Figura. Clase madre
public class Figura {
    private String nombre;

    //constructor por defecto
    public Figura() {
    }

    //constructor con un parámetro
    public Figura(String nombre) {
        this.nombre = nombre;
        System.out.println("Constructor de Figura " + nombre);
    }
   
    //métodos get y set
    public String getNombre() {
        return nombre;
    }
    public void setNombre(String nombre) {
        this.nombre = nombre;
    }

    //método toString()
    //devuelve un String formado por la palabra “Figura” y el nombre de Figura                                
    @Override
    public String toString() {
        return "Figura " + nombre;
    }

}


//Clase Triangulo. Clase derivada de la clase Figura
public class Triangulo extends Figura{

    //constructor con un parámetro
    public Triangulo(String nombre) {
        super(nombre);
        System.out.println("Constructor Triangulo");
    }

    //constructor por defecto
    public Triangulo() {
    }

    //modificación del método toString() para mostrar el mensaje adecuado                                         
    @Override
    public String toString() {
        return super.toString() + " -> Triangulo";
    }
   
}


/Clase Cuadrado. Clase derivada de la clase Figura

    }

    }

    }
}

3.Realizar la estructura de una empresa de transportede  mercancia (personal, vehículos, etc...).


public class Empresa {

    private String nombre;
    private int cantidad;

    public Empresa(String nombre,int cantidad){
        this.nombre = nombre;
        this.cantidad = cantidad;
        
    }

    public String getNombre() {
        return nombre;
    }

    public void setNombre(String nombre) {
        this.nombre = nombre;
    }

    public String getCantidad() {
        return cantidad;
    }

    public void setCantidad(String cantidad) {
        this.cantidad= cantidad;
    }

   
}

public class Vehiculo extends Empresa {

    String  tipo;

    public Vehiculo(String nombre, , int cantidad, String tipo ) {
        super(nombre, cantidad);
        this.tipo = tipo;
    }

    public String getTipo() {
        return Tipo;
    }

    public void setTipo(String tipo) {
        this.tipo = tipo;
    }

    public void show(){
        System.out.println(getNombre() + "-"+getTipo()+"-"+getCantidad());
    }
}
