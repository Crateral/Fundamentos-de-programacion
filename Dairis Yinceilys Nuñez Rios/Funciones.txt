create procedure sp_calculadora
@num1 smallint,
@num2 smallint,
@suma smallint output,
@resta smallint output,
@multi smallint output,
@divis smallint output
as
set @suma=@num1+@num2
set @resta=@num1-num2
set @multi=num1*@num2
set @divis=0
if @num2 <>0
begin
set @divis=@num1/@num2
end
