CREATE FUNCTION fnc_sumar(@num1 int, @num2 int)
returns int 

BEGIN
	DECLARE @resultado int;
	set @resultado = @num1 + @num2 ;
	RETURN @resultado;
end


CREATE FUNCTION fnc_restar(@num1 int, @num2 int)
returns int 

BEGIN
	DECLARE @resultado int;
	set @resultado = @num1 - @num2;
	RETURN @resultado;
end


CREATE FUNCTION fnc_multiplicar (@num1 int, @num2 int)
returns int 

BEGIN
	DECLARE @resultado int;
	set @resultado = @num1 * @num2;
	RETURN @resultado;
end

CREATE PROCEDURE sp_resultados

BEGIN
